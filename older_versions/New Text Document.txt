<script>
// ---------- SCREEN NAVIGATION ----------
function next(id) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById(id).classList.add('active');

    if (id === 'level2') moveKnightTo(60);
    if (id === 'level3') moveKnightTo(50);
    if (id === 'treasure') moveKnightTo(40);
}

// ---------- MUSIC ----------
function startMusic() {
    const music = document.getElementById('bgMusic');
    if (music.paused) music.play();
}

// ---------- KNIGHT MOVEMENT WITH WALKING ----------
function moveKnightTo(percentLeft) {
    const knight = document.getElementById('knight');
    knight.style.left = percentLeft + "%";
    knight.style.animation = "walkBounce 0.5s ease-in-out 4";
    setTimeout(()=>{ knight.style.animation = ""; }, 2000);
}

// ---------- LEVEL 2 MARCH ----------
let marchCount = 0;
const marchLines = [
    "Still here.",
    "Didn‚Äôt leave.",
    "Chose you again.",
    "Again.",
    "Again."
];
function march() {
    if (marchCount < marchLines.length) {
        document.getElementById("marchText").innerText = marchLines[marchCount];
        marchCount++;
    } else {
        next('level3');
    }
}

// ---------- FAIL ----------
function fail() {
    alert("The knight knows this is not the true path.");
}

// ---------- NO BUTTON ----------
const noBtn = document.getElementById("no");
noBtn.style.left = "55%";
noBtn.style.top = "70%";
document.addEventListener("mousemove", e => {
    if (!document.getElementById("treasure").classList.contains("active")) return;
    const rect = noBtn.getBoundingClientRect();
    const dx = e.clientX - (rect.left + rect.width / 2);
    const dy = e.clientY - (rect.top + rect.height / 2);
    if (Math.sqrt(dx*dx + dy*dy) < 120) {
        noBtn.style.left = Math.random() * (window.innerWidth - 100) + "px";
        noBtn.style.top = Math.random() * (window.innerHeight - 50) + "px";
    }
});

// ---------- YES BUTTON ----------
function yes() {
    // Live confetti/fireworks
    for(let i=0;i<150;i++){
        const div=document.createElement('div');
        div.classList.add('confetti');
        div.style.background=['#FFD700','#FF4500','#00CED1','#ADFF2F'][Math.floor(Math.random()*4)];
        div.style.left=Math.random()*window.innerWidth+'px';
        div.style.top=Math.random()*window.innerHeight+'px';
        div.style.width=(5+Math.random()*5)+'px';
        div.style.height=(5+Math.random()*5)+'px';
        document.body.appendChild(div);
        setTimeout(()=>div.remove(),3000+Math.random()*2000);
    }

    // Show success message live
    const treasureScreen = document.getElementById('treasure');
    const msg = document.createElement('p');
    msg.style.fontSize="1.5rem";
    msg.style.fontWeight="bold";
    msg.style.marginTop="20px";
    msg.innerText="The knight has chosen wisely! Quest complete. Return home, hero.";
    treasureScreen.appendChild(msg);
}
